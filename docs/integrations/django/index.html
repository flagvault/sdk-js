Django Integration Guide

This guide shows how to integrate FlagVault with your Django application for feature flag management.

Installation​

pip install flagvault-sdk python-dotenv

Configuration​

1. Environment Variables​

Create a .env file in your project root:

# FlagVault Configuration
FLAGVAULT_API_KEY=test_your_api_key_here

2. Django Settings​

Update your settings.py:

import os
from dotenv import load_dotenv

# Load environment variables
load_dotenv()

# FlagVault Configuration
FLAGVAULT_API_KEY = os.getenv('FLAGVAULT_API_KEY')

Service Integration​

1. Create a FlagVault Service​

Create services.py in your Django app:

import logging
from typing import Dict, Any, Optional
from django.conf import settings
from flagvault_sdk import FlagVaultSDK

logger = logging.getLogger(__name__)


class FlagVaultService:
    """Service class for FlagVault integration"""
    
    _instance = None
    _client = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super(FlagVaultService, cls).__new__(cls)
        return cls._instance
    
    def __init__(self):
        if self._client is None:
            self._initialize_client()
    
    def _initialize_client(self):
        """Initialize FlagVault client with Django settings"""
        api_key = getattr(settings, 'FLAGVAULT_API_KEY', None)
        
        if not api_key:
            logger.warning("FLAGVAULT_API_KEY not configured. Feature flags will use default values.")
            self._client = None
            return
        
        try:
            self._client = FlagVaultSDK(api_key=api_key)
            logger.info("FlagVault client initialized successfully")
        except Exception as e:
            logger.error(f"Failed to initialize FlagVault client: {e}")
            self._client = None
    
    def is_enabled(self, flag_key: str, default_value: bool = False, user_context: Optional[Dict[str, Any]] = None) -> bool:
        """
        Check if a feature flag is enabled
        
        Args:
            flag_key: The feature flag key
            default_value: Default value if flag cannot be retrieved
            user_context: User context for targeted flags (future feature)
        
        Returns:
            Boolean indicating if flag is enabled
        """
        if not self._client:
            logger.debug(f"FlagVault client not available, returning default value {default_value} for flag '{flag_key}'")
            return default_value
        
        # SDK v1.1.0+ handles errors gracefully
        return self._client.is_enabled(flag_key, default_value)


# Global instance
flagvault = FlagVaultService()

2. Use in Views​

from django.shortcuts import render
from .services import flagvault


def my_view(request):
    # Check if new dashboard is enabled
    if flagvault.is_enabled('new-dashboard-layout'):
        return render(request, 'new_dashboard.html')
    else:
        return render(request, 'old_dashboard.html')


def checkout_view(request):
    context = {
        'enhanced_checkout': flagvault.is_enabled('enhanced-checkout', default_value=False),
        'express_payment': flagvault.is_enabled('express-payment', default_value=False),
    }
    return render(request, 'checkout.html', context)

Template Integration​

1. Create Template Tags​

Create templatetags/flagvault_tags.py:

from django import template
from ..services import flagvault

register = template.Library()


@register.simple_tag
def flag_enabled(flag_key, default=False):
    """
    Template tag to check if a feature flag is enabled
    
    Usage:
        {% load flagvault_tags %}
        {% flag_enabled 'new-feature' as is_enabled %}
    """
    return flagvault.is_enabled(flag_key, default)


@register.filter
def is_flag_enabled(flag_key, default=False):
    """
    Template filter to check if a feature flag is enabled
    
    Usage:
        {% if 'new-feature'|is_flag_enabled %}
    """
    return flagvault.is_enabled(flag_key, default)

2. Use in Templates​

{% load flagvault_tags %}

<!-- Using the simple tag -->
{% flag_enabled 'new-dashboard-layout' as new_dashboard %}
{% if new_dashboard %}
    <div class="modern-dashboard">
        <h1>Welcome to the New Dashboard!</h1>
        <!-- New dashboard content -->
    </div>
{% else %}
    <div class="classic-dashboard">
        <h1>Dashboard</h1>
        <!-- Classic dashboard content -->
    </div>
{% endif %}

<!-- Using the filter -->
{% if 'enhanced-checkout'|is_flag_enabled %}
    <button class="btn-primary">Express Checkout</button>
{% else %}
    <button class="btn-default">Proceed to Checkout</button>
{% endif %}

<!-- With default value -->
{% flag_enabled 'beta-features' True as show_beta %}
{% if show_beta %}
    <div class="beta-banner">
        🚀 Beta features are available!
    </div>
{% endif %}

Middleware Integration​

Create middleware to add feature flags to request context:

# middleware.py
from .services import flagvault


class FlagVaultMiddleware:
    """Add feature flags to request object"""
    
    def __init__(self, get_response):
        self.get_response = get_response
    
    def __call__(self, request):
        # Add feature flags to request
        request.feature_flags = flagvault
        
        response = self.get_response(request)
        return response


# In settings.py, add to MIDDLEWARE:
MIDDLEWARE = [
    # ... other middleware
    'myapp.middleware.FlagVaultMiddleware',
]

Use in views:

def my_view(request):
    if request.feature_flags.is_enabled('new-feature'):
        # New feature code
        pass

Context Processor​

Add feature flags to all templates:

# context_processors.py
from .services import flagvault


def feature_flags(request):
    """Add feature flags to template context"""
    return {
        'feature_flags': flagvault,
    }


# In settings.py:
TEMPLATES = [
    {
        'OPTIONS': {
            'context_processors': [
                # ... other processors
                'myapp.context_processors.feature_flags',
            ],
        },
    },
]

Use in any template:

{% if feature_flags.is_enabled('new-feature') %}
    <!-- Feature enabled content -->
{% endif %}

Best Practices​

1. Centralized Flag Management​

Create a flags module for consistency:

# flags.py
class FeatureFlags:
    """Centralized feature flag definitions"""
    # Dashboard
    NEW_DASHBOARD = 'new-dashboard-layout'
    DARK_MODE = 'dark-mode'
    
    # E-commerce
    ENHANCED_CHECKOUT = 'enhanced-checkout'
    EXPRESS_PAYMENT = 'express-payment'
    PRODUCT_REVIEWS = 'product-reviews'
    
    # Beta features
    AI_RECOMMENDATIONS = 'ai-recommendations'
    ADVANCED_SEARCH = 'advanced-search'


# Usage
from .flags import FeatureFlags
from .services import flagvault

if flagvault.is_enabled(FeatureFlags.NEW_DASHBOARD):
    # ...

2. Feature Flag Decorators​

Create decorators for view-level control:

# decorators.py
from functools import wraps
from django.http import Http404
from django.shortcuts import redirect
from .services import flagvault


def feature_required(flag_key, default=False, redirect_url=None):
    """Decorator to require a feature flag to be enabled"""
    def decorator(view_func):
        @wraps(view_func)
        def wrapped(request, *args, **kwargs):
            if not flagvault.is_enabled(flag_key, default):
                if redirect_url:
                    return redirect(redirect_url)
                raise Http404("Feature not available")
            return view_func(request, *args, **kwargs)
        return wrapped
    return decorator


# Usage
@feature_required('beta-features', redirect_url='/upgrade')
def beta_view(request):
    return render(request, 'beta_features.html')

3. Admin Integration​

Show feature flag status in Django admin:

# admin.py
from django.contrib import admin
from django.utils.html import format_html
from .services import flagvault


class FeatureFlagAdmin(admin.ModelAdmin):
    """Base admin class with feature flag info"""
    
    def get_list_display(self, request):
        """Add feature flag status to list display"""
        list_display = super().get_list_display(request)
        if flagvault.is_enabled('admin-enhancements'):
            return list(list_display) + ['feature_status']
        return list_display
    
    def feature_status(self, obj):
        """Show relevant feature flags for this object"""
        flags = []
        if flagvault.is_enabled('enhanced-editing'):
            flags.append('✏️ Enhanced Editing')
        if flagvault.is_enabled('bulk-operations'):
            flags.append('📦 Bulk Operations')
        
        if flags:
            return format_html('<br>'.join(flags))
        return '-'
    feature_status.short_description = 'Active Features'

Testing with Feature Flags​

1. Mock Feature Flags in Tests​

# tests.py
from unittest.mock import patch
from django.test import TestCase
from .services import flagvault


class MyViewTests(TestCase):
    @patch.object(flagvault, 'is_enabled')
    def test_new_dashboard_enabled(self, mock_is_enabled):
        # Mock flag as enabled
        mock_is_enabled.return_value = True
        
        response = self.client.get('/dashboard/')
        self.assertContains(response, 'Welcome to the New Dashboard')
    
    @patch.object(flagvault, 'is_enabled')
    def test_new_dashboard_disabled(self, mock_is_enabled):
        # Mock flag as disabled
        mock_is_enabled.return_value = False
        
        response = self.client.get('/dashboard/')
        self.assertContains(response, 'Dashboard')
        self.assertNotContains(response, 'New Dashboard')

2. Test Both States​

def test_feature_flag_states():
    """Test both enabled and disabled states"""
    from django.test import override_settings
    
    # Test with flags disabled (no API key)
    with override_settings(FLAGVAULT_API_KEY=None):
        assert not flagvault.is_enabled('any-flag')
    
    # Test with custom defaults
    assert flagvault.is_enabled('missing-flag', default_value=True) == True

Production Considerations​

1. Caching​

Cache flag values to reduce API calls:

from django.core.cache import cache

class CachedFlagVaultService(FlagVaultService):
    def is_enabled(self, flag_key: str, default_value: bool = False, user_context=None) -> bool:
        cache_key = f'flagvault:{flag_key}'
        cached_value = cache.get(cache_key)
        
        if cached_value is not None:
            return cached_value
        
        value = super().is_enabled(flag_key, default_value, user_context)
        cache.set(cache_key, value, timeout=300)  # Cache for 5 minutes
        
        return value

2. Performance Monitoring​

import time
from django.core.signals import request_started, request_finished

class FlagVaultMetrics:
    def __init__(self):
        self.flag_checks = {}
        
    def record_check(self, flag_key, duration):
        if flag_key not in self.flag_checks:
            self.flag_checks[flag_key] = []
        self.flag_checks[flag_key].append(duration)
    
    def get_stats(self):
        stats = {}
        for flag, durations in self.flag_checks.items():
            stats[flag] = {
                'count': len(durations),
                'avg_ms': sum(durations) / len(durations) * 1000
            }
        return stats

metrics = FlagVaultMetrics()

3. Graceful Degradation​

The SDK v1.1.0+ automatically handles errors gracefully:

# No need for try/catch - SDK returns default_value on error
is_enabled = flagvault.is_enabled('my-feature', default_value=False)

# Errors are logged but won't crash your application
# Example console output:
# FlagVault: Failed to connect to API for flag 'my-feature', using default: False

Troubleshooting​

Common Issues​

  1. Feature flags always return default value

    • Check if FLAGVAULT_API_KEY is set correctly
    • Verify the API key prefix matches your environment
    • Check Django logs for initialization errors

  2. Template tags not working

    • Ensure the app is in INSTALLED_APPS
    • Check that __init__.py exists in templatetags directory
    • Restart Django server after adding template tags

  3. Slow response times

    • Implement caching as shown above
    • Consider increasing the SDK timeout
    • Check network connectivity to FlagVault API

Debug Mode​

Enable detailed logging:

# settings.py
LOGGING = {
    'version': 1,
    'disable_existing_loggers': False,
    'handlers': {
        'console': {
            'class': 'logging.StreamHandler',
        },
    },
    'loggers': {
        'myapp.services': {
            'handlers': ['console'],
            'level': 'DEBUG',
        },
    },
}

Example: Complete Integration​

Here's a complete example showing all the pieces together:

# views.py
from django.shortcuts import render
from django.contrib.auth.decorators import login_required
from .services import flagvault
from .decorators import feature_required
from .flags import FeatureFlags


@login_required
def dashboard_view(request):
    context = {
        'user': request.user,
        'show_analytics': flagvault.is_enabled(FeatureFlags.ADVANCED_ANALYTICS),
        'dark_mode_available': flagvault.is_enabled(FeatureFlags.DARK_MODE),
    }
    
    if flagvault.is_enabled(FeatureFlags.NEW_DASHBOARD):
        template = 'dashboard/new.html'
    else:
        template = 'dashboard/classic.html'
    
    return render(request, template, context)


@feature_required(FeatureFlags.BETA_FEATURES)
def beta_features_view(request):
    """This view is only accessible when beta features are enabled"""
    return render(request, 'beta/features.html')
<!-- templates/dashboard/new.html -->
{% load flagvault_tags %}

<div class="dashboard-container">
    <h1>Welcome, {{ user.username }}!</h1>
    
    {% if show_analytics %}
    <div class="analytics-panel">
        <!-- Advanced analytics content -->
    </div>
    {% endif %}
    
    {% flag_enabled 'ai-recommendations' as show_ai %}
    {% if show_ai %}
    <div class="ai-recommendations">
        <h2>🤖 AI Recommendations</h2>
        <!-- AI-powered content -->
    </div>
    {% endif %}
