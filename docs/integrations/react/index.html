React Integration Guide

This guide shows how to integrate FlagVault with your React application for feature flag management.

Installation​

npm install @flagvault/sdk
# or
yarn add @flagvault/sdk

Basic Setup​

1. Environment Configuration​

Add your FlagVault API key to your environment variables:

# .env.local (for Next.js) or .env (for Create React App)
REACT_APP_FLAGVAULT_API_KEY=live_your_api_key_here

2. Initialize SDK​

Create a service to manage the FlagVault SDK:

// src/services/flagvault.ts
import FlagVaultSDK from '@flagvault/sdk';

// Initialize SDK once
const sdk = new FlagVaultSDK({
  apiKey: process.env.REACT_APP_FLAGVAULT_API_KEY! // Should be 'live_...' or 'test_...'
});

export default sdk;

Built-in React Hooks​

FlagVault SDK v1.1.0+ includes built-in React hooks for seamless integration. No need to create custom hooks!

1. Basic Feature Flag Hook​

Use the built-in useFeatureFlag hook:

// Import the built-in hook
import FlagVaultSDK, { useFeatureFlag } from '@flagvault/sdk';
import sdk from '../services/flagvault';

function MyComponent() {
  const { isEnabled, isLoading, error } = useFeatureFlag(sdk, 'new-feature', false);

  if (isLoading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;

  return isEnabled ? <NewFeature /> : <OldFeature />;
}

2. Cached Feature Flag Hook​

For better performance, use the built-in useFeatureFlagCached hook:

// Import the built-in cached hook
import FlagVaultSDK, { useFeatureFlagCached } from '@flagvault/sdk';
import sdk from '../services/flagvault';

function MyComponent() {
  const { isEnabled, isLoading, error } = useFeatureFlagCached(
    sdk, 
    'new-feature', 
    false, 
    300000 // 5 minutes cache
  );

  return isEnabled ? <NewFeature /> : <OldFeature />;
}

Component Usage Examples​

1. Basic Component​

// src/components/MyFeature.tsx
import React from 'react';
import { useFeatureFlag } from '@flagvault/sdk';
import sdk from '../services/flagvault';

function MyFeature() {
  const { isEnabled, isLoading, error } = useFeatureFlag(sdk, 'new-ui-feature', false);

  if (isLoading) return <div>Loading...</div>;
  if (error) return <div>Error checking feature flag: {error.message}</div>;

  return isEnabled ? (
    <div className="new-feature">
      <h2>🚀 New Feature UI</h2>
      <p>This is the enhanced user interface!</p>
    </div>
  ) : (
    <div className="classic-feature">
      <h2>📊 Classic UI</h2>
      <p>This is the stable user interface.</p>
    </div>
  );
}

export default MyFeature;

2. Conditional Rendering​

// src/components/Dashboard.tsx
import React from 'react';
import { useFeatureFlag } from '@flagvault/sdk';
import sdk from '../services/flagvault';
import AdvancedAnalytics from './AdvancedAnalytics';
import BasicAnalytics from './BasicAnalytics';

function Dashboard() {
  const { isEnabled: showAdvancedAnalytics } = useFeatureFlag(sdk, 'advanced-analytics', false);
  const { isEnabled: showBetaFeatures } = useFeatureFlag(sdk, 'beta-features', false);
  const { isEnabled: showDarkMode } = useFeatureFlag(sdk, 'dark-mode', false);

  return (
    <div className={`dashboard ${showDarkMode ? 'dark-theme' : 'light-theme'}`}>
      <h1>Dashboard</h1>
      
      {/* Conditional analytics */}
      {showAdvancedAnalytics ? <AdvancedAnalytics /> : <BasicAnalytics />}
      
      {/* Beta features section */}
      {showBetaFeatures && (
        <div className="beta-section">
          <h2>🧪 Beta Features</h2>
          <p>These features are in beta testing.</p>
        </div>
      )}
      
      {/* Theme toggle */}
      {showDarkMode && (
        <button onClick={() => document.body.classList.toggle('dark-theme')}>
          🌓 Toggle Theme
        </button>
      )}
    </div>
  );
}

export default Dashboard;

3. Feature Flag Higher-Order Component​

// src/hocs/withFeatureFlag.tsx
import React from 'react';
import { useFeatureFlag } from '@flagvault/sdk';
import sdk from '../services/flagvault';

interface WithFeatureFlagProps {
  flagKey: string;
  defaultValue?: boolean;
  fallback?: React.ComponentType;
  loading?: React.ComponentType;
}

function withFeatureFlag<P extends object>(
  Component: React.ComponentType<P>,
  options: WithFeatureFlagProps
) {
  return function WrappedComponent(props: P) {
    const { isEnabled, isLoading, error } = useFeatureFlag(
      sdk, 
      options.flagKey, 
      options.defaultValue ?? false
    );

    if (isLoading && options.loading) {
      const LoadingComponent = options.loading;
      return <LoadingComponent />;
    }

    if (isLoading) {
      return <div>Loading feature...</div>;
    }

    if (error || !isEnabled) {
      if (options.fallback) {
        const FallbackComponent = options.fallback;
        return <FallbackComponent />;
      }
      return null;
    }

    return <Component {...props} />;
  };
}

// Usage
const EnhancedCheckout = withFeatureFlag(NewCheckoutComponent, {
  flagKey: 'enhanced-checkout',
  defaultValue: false,
  fallback: StandardCheckoutComponent,
});

Context Provider Pattern​

For app-wide feature flag management:

// src/contexts/FeatureFlagContext.tsx
import React, { createContext, useContext, ReactNode } from 'react';
import { useFeatureFlag } from '@flagvault/sdk';
import sdk from '../services/flagvault';

interface FeatureFlagContextType {
  flags: {
    [key: string]: boolean;
  };
  isLoading: boolean;
}

const FeatureFlagContext = createContext<FeatureFlagContextType | undefined>(undefined);

interface FeatureFlagProviderProps {
  children: ReactNode;
  flagKeys: string[];
}

export function FeatureFlagProvider({ children, flagKeys }: FeatureFlagProviderProps) {
  // This is a simplified example - in practice, you'd want to batch these calls
  const flagResults = flagKeys.map(key => ({
    key,
    ...useFeatureFlag(sdk, key, false)
  }));

  const flags = flagResults.reduce((acc, { key, isEnabled }) => {
    acc[key] = isEnabled;
    return acc;
  }, {} as { [key: string]: boolean });

  const isLoading = flagResults.some(result => result.isLoading);

  return (
    <FeatureFlagContext.Provider value={{ flags, isLoading }}>
      {children}
    </FeatureFlagContext.Provider>
  );
}

export function useFeatureFlags() {
  const context = useContext(FeatureFlagContext);
  if (context === undefined) {
    throw new Error('useFeatureFlags must be used within a FeatureFlagProvider');
  }
  return context;
}

// Usage in App.tsx
function App() {
  return (
    <FeatureFlagProvider flagKeys={['new-ui', 'dark-mode', 'beta-features']}>
      <Dashboard />
    </FeatureFlagProvider>
  );
}

Next.js Integration​

1. API Route for Server-Side Flags​

// pages/api/feature-flags/[flag].ts
import type { NextApiRequest, NextApiResponse } from 'next';
import FlagVaultSDK from '@flagvault/sdk';

const sdk = new FlagVaultSDK({
  apiKey: process.env.FLAGVAULT_API_KEY!
});

export default async function handler(
  req: NextApiRequest,
  res: NextApiResponse
) {
  const { flag } = req.query;

  if (typeof flag !== 'string') {
    return res.status(400).json({ error: 'Invalid flag key' });
  }

  try {
    const isEnabled = await sdk.isEnabled(flag);
    res.status(200).json({ enabled: isEnabled });
  } catch (error) {
    console.error('Error checking feature flag:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
}

2. Server-Side Rendering with Flags​

// pages/dashboard.tsx
import { GetServerSideProps } from 'next';
import FlagVaultSDK from '@flagvault/sdk';

interface DashboardProps {
  features: {
    advancedAnalytics: boolean;
    betaFeatures: boolean;
  };
}

export default function Dashboard({ features }: DashboardProps) {
  return (
    <div>
      <h1>Dashboard</h1>
      {features.advancedAnalytics && <AdvancedAnalytics />}
      {features.betaFeatures && <BetaFeatures />}
    </div>
  );
}

export const getServerSideProps: GetServerSideProps = async () => {
  const sdk = new FlagVaultSDK({
    apiKey: process.env.FLAGVAULT_API_KEY!
  });

  try {
    const [advancedAnalytics, betaFeatures] = await Promise.all([
      sdk.isEnabled('advanced-analytics'),
      sdk.isEnabled('beta-features')
    ]);

    return {
      props: {
        features: {
          advancedAnalytics,
          betaFeatures,
        },
      },
    };
  } catch (error) {
    console.error('Error fetching feature flags:', error);
    return {
      props: {
        features: {
          advancedAnalytics: false,
          betaFeatures: false,
        },
      },
    };
  }
};

Testing with Feature Flags​

1. Mocking Feature Flags in Tests​

// src/__tests__/MyFeature.test.tsx
import React from 'react';
import { render, screen } from '@testing-library/react';
import MyFeature from '../components/MyFeature';
import useFeatureFlag from '../hooks/useFeatureFlag';

// Mock the hook
jest.mock('../hooks/useFeatureFlag');
const mockUseFeatureFlag = useFeatureFlag as jest.MockedFunction<typeof useFeatureFlag>;

describe('MyFeature', () => {
  it('renders new feature when flag is enabled', () => {
    mockUseFeatureFlag.mockReturnValue({
      isEnabled: true,
      isLoading: false,
      error: null,
    });

    render(<MyFeature />);
    expect(screen.getByText('🚀 New Feature UI')).toBeInTheDocument();
  });

  it('renders classic feature when flag is disabled', () => {
    mockUseFeatureFlag.mockReturnValue({
      isEnabled: false,
      isLoading: false,
      error: null,
    });

    render(<MyFeature />);
    expect(screen.getByText('📊 Classic UI')).toBeInTheDocument();
  });

  it('shows loading state', () => {
    mockUseFeatureFlag.mockReturnValue({
      isEnabled: false,
      isLoading: true,
      error: null,
    });

    render(<MyFeature />);
    expect(screen.getByText('Loading...')).toBeInTheDocument();
  });
});

2. Test Utilities​

// src/test-utils/feature-flag-utils.ts
import { ReactElement } from 'react';
import { render } from '@testing-library/react';
import { FeatureFlagProvider } from '../contexts/FeatureFlagContext';

// Custom render with feature flag context
export function renderWithFeatureFlags(
  ui: ReactElement,
  { flagKeys = [], ...renderOptions } = {}
) {
  function Wrapper({ children }: { children: React.ReactNode }) {
    return (
      <FeatureFlagProvider flagKeys={flagKeys}>
        {children}
      </FeatureFlagProvider>
    );
  }

  return render(ui, { wrapper: Wrapper, ...renderOptions });
}

// Mock feature flag responses
export function mockFeatureFlags(flags: Record<string, boolean>) {
  const mockUseFeatureFlag = jest.fn();
  
  mockUseFeatureFlag.mockImplementation((flagKey: string) => ({
    isEnabled: flags[flagKey] || false,
    isLoading: false,
    error: null,
  }));

  return mockUseFeatureFlag;
}

Best Practices​

1. Error Boundaries​

Wrap feature flag components in error boundaries:

// src/components/FeatureFlagErrorBoundary.tsx
import React, { Component, ErrorInfo, ReactNode } from 'react';

interface Props {
  children: ReactNode;
  fallback?: ReactNode;
}

interface State {
  hasError: boolean;
}

class FeatureFlagErrorBoundary extends Component<Props, State> {
  public state: State = {
    hasError: false
  };

  public static getDerivedStateFromError(_: Error): State {
    return { hasError: true };
  }

  public componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error('Feature flag error:', error, errorInfo);
  }

  public render() {
    if (this.state.hasError) {
      return this.props.fallback || <div>Something went wrong with feature flags.</div>;
    }

    return this.props.children;
  }
}

export default FeatureFlagErrorBoundary;

2. Performance Optimization​

// src/hooks/useFeatureFlagOptimized.ts
import { useMemo } from 'react';
import useFeatureFlag from './useFeatureFlag';

// Memoize flag results to prevent unnecessary re-renders
function useFeatureFlagOptimized(flagKey: string, defaultValue = false) {
  const { isEnabled, isLoading, error } = useFeatureFlag(flagKey, defaultValue);
  
  return useMemo(() => ({
    isEnabled,
    isLoading,
    error
  }), [isEnabled, isLoading, error]);
}

export default useFeatureFlagOptimized;

3. Development Tools​

// src/components/FeatureFlagDebugger.tsx (development only)
import React from 'react';
import { useFeatureFlags } from '../contexts/FeatureFlagContext';

function FeatureFlagDebugger() {
  const { flags } = useFeatureFlags();

  if (process.env.NODE_ENV !== 'development') {
    return null;
  }

  return (
    <div style={{
      position: 'fixed',
      bottom: 10,
      right: 10,
      background: '#000',
      color: '#fff',
      padding: 10,
      borderRadius: 5,
      fontSize: 12,
      zIndex: 9999
    }}>
      <h4>🏴 Feature Flags</h4>
      {Object.entries(flags).map(([key, value]) => (
        <div key={key}>
          {key}: {value ? '✅' : '❌'}
        </div>
      ))}
    </div>
  );
}

export default FeatureFlagDebugger;
