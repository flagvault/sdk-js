Flask Integration Guide

This guide shows how to integrate FlagVault with your Flask application for feature flag management.

Installation​

pip install flagvault-sdk python-dotenv

Basic Setup​

1. Environment Configuration​

Create a .env file in your project root:

# FlagVault Configuration
FLAGVAULT_API_KEY=test_your_api_key_here

2. Flask Application Setup​

import os
from flask import Flask, render_template
from dotenv import load_dotenv
from flagvault_sdk import FlagVaultSDK

# Load environment variables
load_dotenv()

app = Flask(__name__)

# Initialize FlagVault
flagvault = FlagVaultSDK(
    api_key=os.getenv('FLAGVAULT_API_KEY')
)

@app.route('/')
def index():
    # Check feature flag
    show_new_ui = flagvault.is_enabled('new-ui', default_value=False)
    
    if show_new_ui:
        return render_template('new_index.html')
    else:
        return render_template('index.html')

Advanced Integration​

1. Flask Extension​

Create a reusable Flask extension:

# flask_flagvault.py
import logging
from flask import current_app, g
from flagvault_sdk import FlagVaultSDK

logger = logging.getLogger(__name__)


class FlagVault:
    """Flask extension for FlagVault integration"""
    
    def __init__(self, app=None):
        self.app = app
        self._client = None
        if app is not None:
            self.init_app(app)
    
    def init_app(self, app):
        """Initialize the extension with Flask app"""
        app.config.setdefault('FLAGVAULT_API_KEY', None)
        
        # Initialize client on first request
        app.before_first_request(self._initialize_client)
        
        # Add to app extensions
        app.extensions = getattr(app, 'extensions', {})
        app.extensions['flagvault'] = self
    
    def _initialize_client(self):
        """Initialize FlagVault client"""
        api_key = current_app.config.get('FLAGVAULT_API_KEY')
        
        if not api_key:
            logger.warning("FLAGVAULT_API_KEY not configured. Feature flags will use default values.")
            return
        
        try:
            self._client = FlagVaultSDK(api_key=api_key)
            logger.info("FlagVault client initialized successfully")
        except Exception as e:
            logger.error(f"Failed to initialize FlagVault client: {e}")
            self._client = None
    
    def is_enabled(self, flag_key, default_value=False):
        """Check if a feature flag is enabled"""
        if not self._client:
            return default_value
        
        return self._client.is_enabled(flag_key, default_value)
    
    @property
    def client(self):
        """Get the underlying FlagVault client"""
        return self._client


# Usage in app.py
from flask import Flask
from flask_flagvault import FlagVault

app = Flask(__name__)
app.config.from_object('config')

# Initialize extension
flagvault = FlagVault(app)

# Or lazy initialization
flagvault = FlagVault()
# Later...
flagvault.init_app(app)

Template Integration​

1. Template Context Processor​

Make feature flags available in all templates:

# app.py
from flask import Flask
from flask_flagvault import FlagVault

app = Flask(__name__)
flagvault = FlagVault(app)

@app.context_processor
def inject_feature_flags():
    """Inject feature flags into template context"""
    return {
        'feature_flags': flagvault,
        'flag_enabled': flagvault.is_enabled
    }

Route Decorators​

1. Feature Required Decorator​

# decorators.py
from functools import wraps
from flask import abort, redirect, url_for, current_app


def feature_required(flag_key, default=False, redirect_to=None):
    """Decorator to require a feature flag to be enabled"""
    def decorator(f):
        @wraps(f)
        def decorated_function(*args, **kwargs):
            flagvault = current_app.extensions.get('flagvault')
            if not flagvault or not flagvault.is_enabled(flag_key, default):
                if redirect_to:
                    return redirect(url_for(redirect_to))
                abort(404)  # Feature not found
            return f(*args, **kwargs)
        return decorated_function
    return decorator


def feature_switch(flag_key, enabled_func, disabled_func=None):
    """Decorator to switch between functions based on feature flag"""
    def decorator(f):
        @wraps(f)
        def decorated_function(*args, **kwargs):
            flagvault = current_app.extensions.get('flagvault')
            if flagvault and flagvault.is_enabled(flag_key):
                return enabled_func(*args, **kwargs)
            elif disabled_func:
                return disabled_func(*args, **kwargs)
            else:
                return f(*args, **kwargs)
        return decorated_function
    return decorator

2. Usage Examples​

from flask import Flask, render_template
from decorators import feature_required, feature_switch

app = Flask(__name__)
flagvault = FlagVault(app)


@app.route('/beta')
@feature_required('beta-features', redirect_to='index')
def beta_features():
    """This route is only accessible when beta features are enabled"""
    return render_template('beta.html')


@app.route('/checkout')
@feature_switch('express-checkout', 
                enabled_func=lambda: render_template('express_checkout.html'),
                disabled_func=lambda: render_template('standard_checkout.html'))
def checkout():
    """This route switches between checkout experiences"""
    pass


@app.route('/api/data')
def api_data():
    """API endpoint with feature flags"""
    data = {
        'version': '1.0',
        'features': {}
    }
    
    if flagvault.is_enabled('api-v2'):
        data['version'] = '2.0'
        data['features']['pagination'] = True
        data['features']['filtering'] = True
    
    return jsonify(data)

Blueprint Integration​

Use feature flags with Flask blueprints:

# blueprints/admin.py
from flask import Blueprint, render_template, current_app

admin_bp = Blueprint('admin', __name__, url_prefix='/admin')

@admin_bp.before_request
def check_admin_features():
    """Check if admin panel is enabled"""
    flagvault = current_app.extensions.get('flagvault')
    if not flagvault or not flagvault.is_enabled('admin-panel'):
        abort(404)


@admin_bp.route('/')
def index():
    flagvault = current_app.extensions.get('flagvault')
    
    context = {
        'advanced_tools': flagvault.is_enabled('admin-advanced-tools'),
        'bulk_operations': flagvault.is_enabled('admin-bulk-operations'),
        'user_impersonation': flagvault.is_enabled('admin-impersonation')
    }
    
    return render_template('admin/index.html', **context)

API Integration​

1. RESTful API with Feature Flags​

from flask import Flask, jsonify, request
from flask_restful import Api, Resource

app = Flask(__name__)
api = Api(app)
flagvault = FlagVault(app)


class FeatureFlagResource(Resource):
    """API endpoint to check feature flags"""
    
    def get(self, flag_key):
        """Check if a feature flag is enabled"""
        is_enabled = flagvault.is_enabled(flag_key)
        return {
            'flag': flag_key,
            'enabled': is_enabled,
            'environment': app.config.get('FLAGVAULT_ENVIRONMENT')
        }


class ProductResource(Resource):
    """Product API with feature flags"""
    
    def get(self, product_id):
        product = get_product(product_id)  # Your product logic
        
        response = {
            'id': product.id,
            'name': product.name,
            'price': product.price
        }
        
        # Add features based on flags
        if flagvault.is_enabled('product-reviews'):
            response['reviews'] = get_product_reviews(product_id)
        
        if flagvault.is_enabled('product-recommendations'):
            response['recommendations'] = get_recommendations(product_id)
        
        return response


api.add_resource(FeatureFlagResource, '/api/flags/<string:flag_key>')
api.add_resource(ProductResource, '/api/products/<int:product_id>')

2. Middleware for Feature Flags​

class FeatureFlagMiddleware:
    """Add feature flag information to requests"""
    
    def __init__(self, app, flagvault):
        self.app = app
        self.flagvault = flagvault
    
    def __call__(self, environ, start_response):
        # Add feature flags to environ
        environ['flagvault'] = self.flagvault
        return self.app(environ, start_response)


# Apply middleware
from werkzeug.middleware.dispatcher import DispatcherMiddleware

app.wsgi_app = FeatureFlagMiddleware(app.wsgi_app, flagvault)

Testing​

1. Mock Feature Flags​

# test_app.py
import pytest
from unittest.mock import Mock, patch
from app import app, flagvault


@pytest.fixture
def client():
    app.config['TESTING'] = True
    with app.test_client() as client:
        yield client


def test_beta_route_enabled(client):
    """Test beta route when feature is enabled"""
    with patch.object(flagvault, 'is_enabled', return_value=True):
        response = client.get('/beta')
        assert response.status_code == 200
        assert b'Beta Features' in response.data


def test_beta_route_disabled(client):
    """Test beta route when feature is disabled"""
    with patch.object(flagvault, 'is_enabled', return_value=False):
        response = client.get('/beta')
        assert response.status_code == 302  # Redirect


class TestDashboard:
    """Test dashboard with different feature flag states"""
    
    def test_new_dashboard_layout(self, client):
        with patch.object(flagvault, 'is_enabled') as mock_is_enabled:
            # Configure mock for specific flags
            def side_effect(flag_key, default=False):
                if flag_key == 'new-dashboard-layout':
                    return True
                return default
            
            mock_is_enabled.side_effect = side_effect
            
            response = client.get('/dashboard')
            assert b'grid-layout' in response.data

Production Best Practices​

1. Caching​

Implement caching to reduce API calls:

from flask_caching import Cache
from datetime import timedelta

cache = Cache(app, config={'CACHE_TYPE': 'simple'})


class CachedFlagVault(FlagVault):
    """FlagVault with caching support"""
    
    def is_enabled(self, flag_key, default_value=False):
        cache_key = f'flagvault:{flag_key}'
        
        # Try to get from cache
        cached_value = cache.get(cache_key)
        if cached_value is not None:
            return cached_value
        
        # Get from API
        value = super().is_enabled(flag_key, default_value)
        
        # Cache for 5 minutes
        cache.set(cache_key, value, timeout=300)
        
        return value
    
    def clear_cache(self, flag_key=None):
        """Clear cached flag values"""
        if flag_key:
            cache.delete(f'flagvault:{flag_key}')
        else:
            # Clear all flag cache
            cache.clear()

2. Performance Monitoring​

import time
from flask import g


@app.before_request
def before_request():
    g.start_time = time.time()
    g.flag_checks = []


@app.after_request
def after_request(response):
    if hasattr(g, 'flag_checks') and g.flag_checks:
        total_time = sum(check['duration'] for check in g.flag_checks)
        response.headers['X-FeatureFlag-Checks'] = str(len(g.flag_checks))
        response.headers['X-FeatureFlag-Time'] = f"{total_time:.3f}s"
    return response


# Wrap is_enabled to track performance
original_is_enabled = flagvault.is_enabled

def tracked_is_enabled(flag_key, default_value=False):
    start = time.time()
    result = original_is_enabled(flag_key, default_value)
    duration = time.time() - start
    
    if hasattr(g, 'flag_checks'):
        g.flag_checks.append({
            'flag': flag_key,
            'duration': duration,
            'result': result
        })
    
    return result

flagvault.is_enabled = tracked_is_enabled

3. Health Check Endpoint​

@app.route('/health')
def health_check():
    """Health check endpoint including feature flag status"""
    health = {
        'status': 'healthy',
        'timestamp': datetime.utcnow().isoformat(),
        'feature_flags': {
            'connected': flagvault.client is not None
        }
    }
    
    # Test a canary flag
    try:
        canary = flagvault.is_enabled('health-check-canary', default_value=True)
        health['feature_flags']['canary'] = canary
    except Exception as e:
        health['feature_flags']['error'] = str(e)
        health['status'] = 'degraded'
    
    return jsonify(health)

Example: Complete Application​

Here's a complete Flask application with FlagVault integration:

# app.py
import os
from flask import Flask, render_template, jsonify
from flask_caching import Cache
from dotenv import load_dotenv
from flask_flagvault import FlagVault
from decorators import feature_required

# Load environment
load_dotenv()

# Create Flask app
app = Flask(__name__)
app.config.from_object('config.Config')

# Initialize extensions
cache = Cache(app, config={'CACHE_TYPE': 'simple'})
flagvault = FlagVault(app)


# Context processor
@app.context_processor
def inject_feature_flags():
    return {
        'flag_enabled': flagvault.is_enabled
    }


# Routes
@app.route('/')
def index():
    features = {
        'hero_banner': flagvault.is_enabled('hero-banner'),
        'testimonials': flagvault.is_enabled('testimonials'),
        'pricing_table': flagvault.is_enabled('pricing-table')
    }
    return render_template('index.html', features=features)


@app.route('/dashboard')
@login_required  # Your auth decorator
def dashboard():
    user_features = {
        'analytics': flagvault.is_enabled('user-analytics'),
        'export': flagvault.is_enabled('data-export'),
        'collaboration': flagvault.is_enabled('team-collaboration')
    }
    
    if flagvault.is_enabled('new-dashboard'):
        template = 'dashboard/v2.html'
    else:
        template = 'dashboard/v1.html'
    
    return render_template(template, features=user_features)


@app.route('/api/features')
def api_features():
    """List all available features for the current user"""
    all_features = [
        'new-dashboard',
        'user-analytics',
        'data-export',
        'team-collaboration',
        'dark-mode',
        'beta-features'
    ]
    
    enabled_features = {
        feature: flagvault.is_enabled(feature)
        for feature in all_features
    }
    
    return jsonify({
        'features': enabled_features,
        'environment': app.config.get('FLAGVAULT_ENVIRONMENT')
    })


@app.route('/beta')
@feature_required('beta-features', redirect_to='index')
def beta():
    return render_template('beta.html')


if __name__ == '__main__':
    app.run(debug=True)
